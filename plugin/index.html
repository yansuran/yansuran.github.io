环形进度图生成
代码片段


     /**
      *  @ 生成 0%-100% 环形进度PNG拼图,动态设置backgroundPosition使用
      */
     window.onload = function(){
          var canvas = document.getElementById("mycanvas"),
              context =  canvas.getContext("2d");
          /* Arc()方法绘制半径为60圆弧 */
          context.beginPath();
          context.arc(61 ,(61+122*i)  , 60  ,1.5*Math.PI , (1.5+0.02*i)*Math.PI );
          context.lineWidth = 1;
          context.strokeStyle = "#1da563";
          context.stroke();
          /* 圆环文本设置 , 不需要可删除 */
          context.font="14px Arial";
          context.fillText( i+"%" ,61, (61+122*i) ); // 圆心百分数 (可选)
          context.textAlign='center';//文本水平对齐方式
          context.textBaseline='middle';//文本垂直方向，基线位置
          };
     };

延时隐藏下拉
代码片段


     /**
      *  @ 用户体验：延时隐藏下拉（防止意外滑出） ,并重置滚动条scrollTop
      */
     var slidebox = $('.scroll'), oHider;
     $(".oBox").hover(function(){
          clearInterval(oHider);
          if(slidebox.is(":hidden")){
             slidebox.show().scrollTop(0);
          }else{
             return false;
          }
      },function(){
          oHider = setTimeout(function(){
             slidebox.hide();
          }, 1000)
     });

页面类目导航
代码片段


     /**
     *  @ 页面类目导航悬浮 , 可视区域导航高亮 , 导航设置锚点
     */
     $(window).scroll(function(){
          var top = $(document).scrollTop(), items = $(".model").find(".ma"),
              fixedMenu = $(".fixedModel"); var curId;
          items.each(function(){
               var itemsTop = $(this).offset().top;
               if(top > itemsTop-100){
                 curId = "#" + $(this).attr("id");
               }else{
                 return false;
               }
          });
          var curLink = fixedMenu.find(".current");
          if( curId && curLink.attr("href") != curId ){
               curLink.removeClass("current");
               fixedMenu.find( "[href=" + curId + "]" ).addClass("current");
          }
     });

阻止鼠标滑轮事件冒泡
代码片段


     /**
     *  @ 阻止鼠标滑轮事件冒泡: 滚动页面中div的滚动条，禁用页面滚动条的滚动
     */
     preventScroll:function(){
          $(this).each(function(){
               var _this = this;
               if(navigator.userAgent.indexOf('Firefox') >= 0){
                    _this.addEventListener('DOMMouseScroll',function(e){
                         _this.scrollTop += e.detail > 0 ? 60 : -60; // 重置scrollTop值
                         e.preventDefault();
                     },false);
               }else{
                    _this.onmousewheel = function(e){
                         e = e || window.event;
                         _this.scrollTop += e.wheelDelta > 0 ? -60 : 60; // 重置scrollTop值
                         return false;
                    };
               }
          })
     }
